mean_x = np.mean(X)
mean_y = np.mean(Y)

m = len(X)
numer = 0
denom = 0
for i in range(m):
    numer +=(X[i] - mean_x)*(Y[i]-mean_y)
    denom += (X[i] - mean_x)**2
b1 = numer / denom
b0 = mean_y - (b1 * mean_x)
